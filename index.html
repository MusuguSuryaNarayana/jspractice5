//Solve Using For-In Loop

//Q-1
var car = 
{
  brand: 'Toyota',
  model: 'Corolla',
  year:   2020
};
for(i in car){
  console.log(`${i} : ${car[i].toString().toUpperCase()}`);
};
OUTPUT:brand : TOYOTA
       model : COROLLA
       year : 2020

//Q-2
var numbers = [1,2,3,4,5];
for(i in numbers)
{
  console.log(`${numbers[i]}-HI`);
};
OUTPUT: 1-HI
        2-HI
        3-HI
        4-HI
        5-HI

//Q-3

{

};

       1-BANANA
       2-CHERRY
       3-DATE
